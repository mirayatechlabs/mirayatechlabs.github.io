// Parallax super-leggero const bg = document.getElementById('bg'); const orbs = document.querySelectorAll('.orb'); let tx=0, ty=0, px=0, py=0; function loop(){ px += (tx - px) * 0.08; py += (ty - py) * 0.08; const bgX = px * 14, bgY = py * 8; bg.style.transform = `translate(${bgX}%, ${bgY}%) scale(1.08)`; orbs.forEach((o,i)=>{ const k=(i+1)*8; o.style.transform = `translate(${px*k}px, ${py*k}px)`; }); requestAnimationFrame(loop); } requestAnimationFrame(loop); window.addEventListener('pointermove', e=>{ const w = innerWidth, h = innerHeight; tx = (e.clientX/w)*2 - 1; ty = (e.clientY/h)*2 - 1; }, {passive:true}); window.addEventListener('deviceorientation', e=>{ const gx=(e.gamma||0)/30, gy=(e.beta||0)/45; tx = Math.max(-1, Math.min(1, gx)); ty = Math.max(-1, Math.min(1, gy)); }, {passive:true});
