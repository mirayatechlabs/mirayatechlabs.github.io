### src/i18n/utils.ts

Creare il file. Funzione `useTranslations(lang: string)` che importa il JSON corretto e lo ritorna come oggetto tipizzato. Nessuna dipendenza esterna.

```typescript
// utils.ts — helper i18n per caricare le stringhe nella lingua corrente
// Importato dai componenti Astro con: const t = useTranslations(Astro.currentLocale ?? 'it')

import it from './it.json';
import en from './en.json';

// Tipo inferito dai file JSON — TypeScript rileva automaticamente la struttura
type Translations = typeof it;

/**
 * Ritorna le stringhe UI per la lingua specificata.
 * Usa 'it' come fallback per qualsiasi lingua non supportata.
 */
export function useTranslations(lang: string): Translations {
  if (lang === 'en') return en as Translations;
  return it;
}
```
