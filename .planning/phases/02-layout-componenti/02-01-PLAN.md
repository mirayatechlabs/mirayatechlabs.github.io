```javascript
// Mapping rotte IT → EN
const itToEn: Record<string, string> = {
  '/': '/en/',
  '/servizi': '/en/services',
  '/servizi/': '/en/services',
  '/configuratore': '/en/configurator',
  '/configuratore/': '/en/configurator',
  '/contatti': '/en/contact',
  '/contatti/': '/en/contact',
};
// Mapping rotte EN → IT
const enToIt: Record<string, string> = {
  '/en/': '/',
  '/en/services': '/servizi',
  '/en/services/': '/servizi',
  '/en/configurator': '/configuratore',
  '/en/configurator/': '/configuratore',
  '/en/contact': '/contatti',
  '/en/contact/': '/contatti',
};
const currentPath = Astro.url.pathname;
const altLangHref = lang === 'it'
  ? (itToEn[currentPath] ?? '/en/')
  : (enToIt[currentPath] ?? '/');
const altLangLabel = lang === 'it' ? 'EN' : 'IT';
const currentLangLabel = lang === 'it' ? 'IT' : 'EN';
```

```javascript
// Hamburger menu — toggle classe nav-open su <nav>
const nav = document.getElementById('main-nav');
const hamburger = nav?.querySelector('[aria-label="Menu"]');
if (nav && hamburger) {
  hamburger.addEventListener('click', () => {
    const isOpen = nav.classList.toggle('nav-open');
    hamburger.setAttribute('aria-expanded', String(isOpen));
  });
}
```
