<interfaces>
Tailwind CSS 4 con Astro richiede @tailwindcss/vite come plugin Vite diretto.
NON usare @astrojs/tailwind (deprecato, solo per Tailwind 3).

astro.config.mjs pattern corretto:

  import { defineConfig } from 'astro/config';
  import tailwindcss from '@tailwindcss/vite';
  import sitemap from '@astrojs/sitemap';

  export default defineConfig({
    output: 'static',
    site: 'https://mirayatechlabs.github.io',
    base: '/',
    integrations: [sitemap()],
    vite: { plugins: [tailwindcss()] },
    i18n: {
      defaultLocale: 'it',
      locales: ['it', 'en'],
      routing: { prefixDefaultLocale: false },
    },
  });

global.css pattern per Tailwind 4 (NON usare @tailwind directives):
  @import "tailwindcss";
  @theme { /* variabili colore */ }
</interfaces>

FILE 1: astro.config.mjs

  // Configurazione Astro 5 — mirayatechlabs.github.io
  // Deploy: GitHub Pages statico | i18n: IT default, EN su /en/
  import { defineConfig } from 'astro/config';
  import tailwindcss from '@tailwindcss/vite';
  import sitemap from '@astrojs/sitemap';

  export default defineConfig({
    output: 'static',
    site: 'https://mirayatechlabs.github.io',
    base: '/',
    integrations: [
      sitemap({
        i18n: {
          defaultLocale: 'it',
          locales: {
            it: 'it-IT',
            en: 'en-US',
          },
        },
      }),
    ],
    vite: {
      plugins: [tailwindcss()],
    },
    i18n: {
      defaultLocale: 'it',
      locales: ['it', 'en'],
      routing: {
        prefixDefaultLocale: false,
      },
    },
  });
